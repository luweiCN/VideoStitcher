```typescript
export async function runCinematographerAgent(
  input: CinematographerInput,
  provider: AIProvider,
  logger?: Logger
): Promise<CinematographerResult> {
  // 自动检测工作流模式
  const useMultiStage = !input.modelCapabilities.supportsReferenceImage;

  if (useMultiStage) {
    return runMultiStage(input, provider, logger);
  } else {
    return runSingleStage(input, provider, logger);
  }
}
```
