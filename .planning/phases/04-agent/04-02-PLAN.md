```typescript
interface CinematographerPlannerInput {
  storyboardOutput: StoryboardOutput;  // 分镜输出（含 25 帧描述）

  modelCapabilities: ModelCapabilities; // 模型能力配置
}
```

### Planner 输出
```typescript
interface RenderPlan {
  workflowMode: 'mode_a' | 'mode_b';
  totalChunks: number;
  chunks: RenderChunk[];
  modelSpecificNotes: string;
}

interface RenderChunk {
  chunkId: number;
  durationSeconds: number;
  startFrameIndex: number;
  endFrameIndex: number;
  firstFrameIndex: number;
  lastFrameIndex: number;
  cameraMovement: string;
  transitionNote: string;
  promptContext: string;
}
```

### Executor 输入
```typescript
interface CinematographerExecutorInput {
  renderPlan: RenderPlan;
  frameImages: FrameImage[];  // 分镜帧图
  modelConfig: ModelConfig;   // 模型配置
}
```

### Executor 输出
```typescript
interface CinematographerExecutorOutput {
  videoSegments: string[];  // 视频片段 URL 数组
  totalDuration: number;
}
```
